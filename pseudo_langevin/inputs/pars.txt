# main inputs

## model
input_dim   100				#dimension of the input vector
hidden_dims [500, 100]		#number of neurons in the hidden layers
output_dim  10				#number of neurons in the output layer (i.e. the number of classes)
model_seed  0				#seed for weights generation
from        None			#path to the .pt file containing the starting weights (if None, gaussian initialization of the weights)


## data
K       10					#number of reference kspin vectors
d       100					#dimension of the kspin vectors
pflip   0.355				#probability of flipping a spin component starting when generating a new kspin vector from a reference one
P_train 70_000				#dimension of the training set (mandatory)
P_val   15_000				#dimension of the validation set (default: 0)
P_test  15_000				#dimension of the test set (default: 0)

## sampler
stime           1.0e+5		#simulated time (moves=stime/dt, with dt=1.0 by default)
T               1.0e-6		#sampled temperature
T_ratio_i       1.0e-2		#initial ratio between the sampled temperatures and the mini-batch effective temperatures
m1              0.1			#mobility to compute the correct mass of the system (where mobility^2=1-exp(-viscosity*dt))
lamda           1.0e-4		#cumulative Lagrange multiplier of the L2 regularization term
T_ratio_max     1.0e-1		#maximum value for the ratio between mini-batch and sampled temperatures, after which the mass matrix (and consequently the momenta) start to get updated in its place
mbs             5_000		#mini-batch fraction (w.r.t. the full batch) employed for the computation of the gradients during the dynamics
bss             0			#batch splitting size used to divide the full batch when sampling

max_extractions 1000		#maximum number of mini-batch extractions during estimate of var
min_extractions 100			#minimum number of mini-batch extractions added to improve the estimate of var
threshold_est   0.1			#threshold for convergence of the var estimate
max_adj_step    50000		#maximum number of steps after which var is adjourned
min_adj_step    5000		#minimum number of steps after which var is adjourned
threshold_adj   0.1			#threshold for keeping the streak during var adjournment
seed            7			#torch generator seed

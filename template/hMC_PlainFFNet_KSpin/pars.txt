# main inputs

## model
input_dim       100				#dimension of the input layer (i.e. the input vector dimension)
hidden_dims     [500, 100]		#list of the number of neurons in each hidden layer
output_dim      10				#number of neurons in the output layer (i.e. the number of classes)
model_seed      0				#seed for weights generation (default: 0)
from            None			#path to the .pt file containing the starting weights. If None, gaussian initialization of the weights (default: None)


## data
K               10				#number of classes (that is, the number of reference binary spin vectors)
d               100				#dimension of the binary spin vectors
pflip           0.355			#probability of flipping a spin component when generating a new binary spin vector from a reference one
P_train         100_000			#number of examples in the training set
P_val           0				#number of examples in the validation set (default: 0)
P_test          0				#number of examples in the test set (default: 0)


## sampler
stime           1.0e+3			#simulated time, equal to the number of moves times the integration time step times the number of integrations per-move
T               1.0e-6			#sampled temperature
dt              1.0				#single-integration time step (default: 1.0)
isteps          100				#number of integration steps per-move (default: 1000)
M               1.0				#mass of of each parameter of the network (default: 1.0)
lamda           0.0				#Lagrange multiplier of the L2 regularization term added on top of the cost function (default: 0.0)
gamma           0.0				#harmonic constant of the quadratic term computed with respect to the simulation starting weight vector added on top of the cost function (default: 0.0)
bss             0				#batch splitting size used to divide the full batch when measuring the observables (loss function, cost function, and error function over all the input datasets) (default: 0)
seed            0				#custom pytorch random-number generator seed (default: 0)

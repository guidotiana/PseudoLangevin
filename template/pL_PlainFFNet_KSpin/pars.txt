								#If a temperatures ratio exceeds this limit, the associated component mass is re-computed and its momentum re-extracted (as during the reset).

max_extractions 1000			#maximum number of mini-batch extractions during estimate of the mini-bath gradient variances (default: 1000)

max_adj_step    50000			#maximum number of integration steps after which the mini-bath gradient variances are adjourned, along with the temperatures ratios (default: 5_000)

seed            0				#custom pytorch random-number generator seed (default: 0)
